# Widget Wrapper

## Usage

<div className="space-y-12">
  <Demo title="Default">
    <WidgetWrapper>
      <div>Content</div>
    </WidgetWrapper>
  </Demo>
</div>


```astro
---
import { WidgetWrapper } from 'yt-astro-ui';
---

<!-- Default -->
<WidgetWrapper>
  <div>Content</div>
</WidgetWrapper>

<!-- With backgound and props -->
<WidgetWrapper
  as="div"
  id="hero"
  containerClass="text-center text-base"
>
  <div>Content</div>
  <Fragment slot="bg">
      <div class="absolute inset-0 bg-primary/10 dark:bg-card"></div>
    </Fragment>
</WidgetWrapper>
```
